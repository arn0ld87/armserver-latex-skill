% alexle135 professional document preamble — \input this after \documentclass.
% Requires: lualatex or xelatex (uses fontspec + system fonts). Do NOT use with pdflatex.
% Fonts: IBM Plex Serif (headings/display) + IBM Plex Sans (body) — matches alexle135.de brand.
% Accent: #FF6A00 (brand orange), used sparingly for rules and heading color.

\usepackage{fontspec}
\usepackage{polyglossia}
\setmainlanguage{german}
\setmainfont{IBM Plex Sans}
\newfontfamily\displayfont{IBM Plex Serif}
\setmonofont{IBM Plex Mono}[Scale=0.9]

\usepackage[a4paper,margin=2.6cm]{geometry}
\usepackage{xcolor}
\definecolor{accent}{HTML}{FF6A00}
\definecolor{ink}{HTML}{1A1A1A}
\definecolor{subtleline}{HTML}{D8D8D8}
\color{ink}

\usepackage{titlesec}
\titleformat{\section}
  {\displayfont\bfseries\Large\color{ink}}{\thesection}{0.6em}{}
  [{\color{accent}\titlerule[1.2pt]}]
\titleformat{\subsection}
  {\displayfont\bfseries\large\color{ink}}{\thesubsection}{0.6em}{}
\titlespacing*{\section}{0pt}{1.6em}{0.8em}
\titlespacing*{\subsection}{0pt}{1.2em}{0.5em}

\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0.4pt}
\fancyfoot[L]{\small\color{gray}alexle135.de}
\fancyfoot[R]{\small\color{gray}\thepage}

\usepackage{enumitem}
\setlist{itemsep=0.2em, topsep=0.4em}

\usepackage{booktabs}
\usepackage{hyperref}
\hypersetup{colorlinks=true, linkcolor=accent, urlcolor=accent, citecolor=accent}

% Title block: use inside \begin{document}, before the content.
% Usage: \docTitle{Titel}{Untertitel (optional)}{Autor/Agent}{Datum}
\newcommand{\docTitle}[4]{%
  \begingroup
  \color{accent}\rule{\linewidth}{2pt}\par\vspace{0.4em}
  \endgroup
  {\displayfont\bfseries\huge\color{ink} #1\par}
  \ifx&#2&\else{\displayfont\Large\color{gray} #2\par}\fi
  \vspace{0.6em}
  {\small\color{gray} #3 \textbullet\ #4\par}
  \vspace{1.4em}
}
